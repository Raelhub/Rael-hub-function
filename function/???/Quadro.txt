local FaseDosQuadros = {}

local Workspace = game:GetService("Workspace")

function FaseDosQuadros.EncontrarPasta()
    for _, QuandoPasta in ipairs(Workspace:GetChildren()) do
        if QuandoPasta:IsA("Folder") and QuandoPasta:FindFirstChild("Jumpscare") then
            print(QuandoPasta.Name)
            return QuandoPasta
        end
    end
    return nil
end

function FaseDosQuadros.TalkKenio(folder)
    for _, folder2 in ipairs(folder:GetChildren()) do
        if folder2:IsA("Folder") and folder2:FindFirstChild("Quest") then
            print(folder2.Name)
            fireproximityprompt(folder2.Quest:FindFirstChildOfClass("ProximityPrompt"))
            return folder2
        end
    end
    return nil
end
