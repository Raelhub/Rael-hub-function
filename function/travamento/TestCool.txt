local GeradoresPosition = {}


function AtualizarGeradores()
    GeradoresPosition = {}
    for index, generator in pairs(workspace.CurrentRoom:GetChildren()[1].Generators:GetChildren()) do
        GeradoresPosition["gerador" .. index] = {
            Name = Main.generator .. " " .. index,
            Function = function()
                if generator and generator:FindFirstChild("TeleportPositions") then
                    local TeleportPosition = generator.TeleportPositions.TeleportPosition.Position
                    return TeleportPosition
                else
                    RaelHubFunction.CreateNotification("The generator " .. index .. " does not exist", 3)
                    return nil
                end
            end
        }
    end
    AtualizarDropdownGeradores()
end

local selectedgerador

Principal:AddSection({Main.section2})

local GeradorDropdown = Principal:AddDropdown({
    Name = Main.dropdowntext,
    Description = "",
    Options = {}, 
    Default = "...",
    Flag = "Teleporters geradores",
    Callback = function(Value)
      for _, Gerador in pairs(GeradoresPosition) do
        if Gerador.Name == Value then
          selectedgerador = Gerador
          break
        end
      end
    end
})

function AtualizarDropdownGeradores()
    local GeradorOptions = {}
    for _, gerador in pairs(GeradoresPosition) do
        table.insert(GeradorOptions, gerador.Name)
    end
    if GeradorDropdown then
      GeradorDropdown:Set(GeradorOptions)
    end
end

  Max = 100,
  Default = 25

PlayerSpeed:Callback(function(Value)
  local ReplicatedStorage = game:GetService("ReplicatedStorage")

  for _, Object in ipairs(ReplicatedStorage:GetDescendants()) do
    if Object.Name == "CheckPlayerSpeed" then
      Object:Destroy()
      break
    end
  end

  JogadorSpeedValue = Value
end)

local ToggleSpeed = Player:AddToggle({
  Name = Jogador.toggle1,
  Description = "",
  Flag = "",
  Default = false
})

ToggleSpeed:Callback(function(Value)
  local RunService = game:GetService("RunService")
  local Player = game.Players.LocalPlayer
  local Humanoid = Player.Character and Player.Character:FindFirstChildOfClass("Humanoid")

  if Value then

    JogadorSpeedRunService = RunService.Heartbeat:Connect(function()
      if Humanoid then
        Humanoid.WalkSpeed = JogadorSpeedValue
      end
    end)
  else

    if JogadorSpeedRunService then
      JogadorSpeedRunService:Disconnect()
      JogadorSpeedRunService = nil
    end
    if Humanoid then
      Humanoid.WalkSpeed = 16
    end
  end
end)
